基于ARM的Linux操作系统引导详解

修改地址转换程序

Linux内核经常需要在虚拟地址和物理地址之间进行转换，在使用内存管理模块之前，虚拟地址和物理地址之间的转换为线性转换，物理地址和虚拟地址之间通常用下面的公式进行转换：

$$ phy = virt + (PHYS\_OFFSET - PAGE\_OFFSE) $$

$$ virt = PHYS - (PHYS\_OFFSET - PAGE\_OFFSET) $$

在配置文件中有一个CONFIG_ARM_PATCH_PHYS_VIRT的编译开关，如果没有设置该编译开关，则虚拟地址和物理地址之间相互转换的函数简单地定义为：

static inline unsigned long __phys_to_virt(phys_addr_t x)

{

    return x - PHYS_OFFSET + PAGE_OFFSET;

}

static inline phys_addr_t __virt_to_phys(unsigned long x)

{

    return (phys_addr_t)x - PAGE_OFFSET + PHYS_OFFSET;

}

这种转换方式只适应1：1的转换，即线性转换。Linux系统中虚拟地址和物理地址不是一对一的映射，即同一个物理地址可能映射到不同的虚拟地址。Linux系统也需要支持多种系统，不同的系统，物理地址所处的页码不同，同时页码在后续的内存管理模块中经常用到，也需要不断更新。因此，不能够通过在编译时设置固定的PHYS_OFFSET实现虚拟地址和物理地址之间的转换，而需要通过其它技巧，实现在运行过程中对不同内存结构进行虚拟地址和物理地址之间的转换。物理地址和虚拟地址之间转换的Linux内核程序为：

static inline unsigned long __phys_to_virt(phys_addr_t x)

{

    unsigned long t;

    __pv_stub((unsigned long) x, t, "sub", __PV_BITS_31_24);

    return t;

}

static inline phys_addr_t __virt_to_phys(unsigned long x)

{

    phys_addr_t t;

    if (sizeof(phys_addr_t) == 4) { /* 当物理地址为32位时 */

        __pv_stub(x, t, "add", __PV_BITS_31_24);

    } else { /* 当物理地址为40位时 */

        __pv_stub_mov_hi(t);

        __pv_add_carry_stub(x, t);

    }

    return t;

}

在编译过程中，__pv_stub由下列的一段代码替换，其中instr为“add”或“sub”汇编指令。

#define __pv_stub(from,to,instr,type)   \

__asm__("@ __pv_stub\n"                 \ @汇编语言注释行

"1: " instr " %0, %1, %2\n"            \ @汇编指令，通常为加或减指令

                                        @%0=from, %1=to, %2=type

" .pushsection .pv_table,\"a\"\n"   \ @切换到pv_table段

" .long 1b\n"              \ @把标号1的地址保存在pv_table段

" .popsection\n" \ @回到当前段

: "=r" (to) \

: "r" (from), "I" (type))

而__pv_stub_mov_hi和__pv_add_carry_stub分别由下列两段代码代换。

#define __pv_stub_mov_hi(t) \

__asm__ volatile("@ __pv_stub_mov\n" \

"1: mov %R0, %1\n" \

" .pushsection .pv_table,\"\a\"\n" \

" .long 1b\n" \

" .popsection\n" \

: "=r" (t) \

: "I" (__PV_BITS_7_0))

#define __pv_add_carry_stub(x, y) \

__asm__ volatile("@ __pv_add_carry_stub\n" \

"1: adds %Q0, %1, %2\n" \

" adc %R0, %R0, #0\n" \

" .pushsection .pv_table,\"\a\"\n" \

" .long 1b\n" \

" .popsection\n" \

: "+r" (y) \

: "r" (x), "I" (__PV_BITS_31_24) \

: "cc")

其中，__pv_stub_mov_hi（经补丁程序修改后）的作用是把物理地址和虚拟地址偏差的高32位存入寄存器对当中的高权重寄存器，__pv_add_carry_stub（经补丁程序修改后）的作用是把偏差的低32位与虚拟地址相加，然后把进位位加到偏差的高32位。在补丁程序对这两段代码修改前，使用了立即数__PV_BITS_31_24和__PV_BITS_7_0。使用这两个立即数，一方面可以使编译程序在编译时能够生成使用立即数的mov指令和adds指令，并设置正确的mov指令和adds指令的rotation域，从而保证指令adds的12位立即数域对应正确的低32位地址偏差。补丁程序可以通过这两条指令的立即数域中的rotation域，辨识指令对应的宏是__pv_stub_mov_hi还是__pv_add_carry_stub，从而对不同的指令进行相应的修补。这些代码的特别之处是把标号1的虚拟地址，即指令地址保存在名为pv_table的段内，而不是保存在当前段内。

上面定义的宏除了实现加或减指令以实现地址之间的转换外，还把指令本身在调用时的地址保存在一个名为__pv_table的表格内。在执行该指令前，补丁程序通过修改指令的立即数部分就可以实现不同的地址转换。在编译程序过程中，所有需要修改的指令的地址都被一一记录在__pv_table中。在宏定义中使用了arm特有的输入、输出限制修饰符R和Q，R限定使用寄存器对中的高权重寄存器，Q限定使用寄存器对中的低权重寄存器。

位于地址__fixup_pv_table处的代码是一段补丁程序，该补丁动态地支持多种内存结构，修补的基本方法就是修改__pv_table中每一个指针指向的指令的立即数，从而实现不同偏差的地址转换。在调用位于__fixup_pv_table的代码之前，寄存器r8保存有PHYS_OFFSET的值。下面列出了补丁程序代码，我们来仔细分析这一子程序的工作流程。

__fixup_pv_table:

adr r0, 1f @计算标号1的物理地址

ldmia r0, {r3-r7}   @把pv_table_begin，__pv_table_end，__pv_phys_pfn_offset，   @__pv_offset等的虚拟地址分别读入r3-r7

mvn ip, #0

subs r3, r0, r3 @ r3：物理地址和虚拟地址之差

add r4, r4, r3 @ 计算__pv_table_begin的物理地址

add r5, r5, r3 @ 计算__pv_table_end的物理地址

add r6, r6, r3 @ 计算__pv_phys_pfn_offset的物理地址

add r7, r7, r3 @ 计算__pv_offset的物理地址

mov r0, r8, lsr #PAGE_SHIFT @ 转换为 PFN，PHYS_OFFSET is in r8

str r0, [r6] @ 保存到__pv_phys_pfn_offset，供MMU使用

strcc ip, [r7, #HIGH_OFFSET] @保存__pv_offset高32位，高32通常为0或-1

mov r6, r3, lsr #24 @ 用于 add/sub指令的常量

teq r3, r6, lsl #24 @ 测试是否为16MiB的整数倍

bne __error

str r3, [r7, #LOW_OFFSET] @ 保存到__pv_offset低端

b __fixup_a_pv_table

ENDPROC(__fixup_pv_table)

.align

1: .long .

.long pf_table_begin

.long __pv_table_end

2: .long __pv_phys_pfn_offset

.long __pv_offset

子程序中第一行代码计算标号1的物理地址，结果保存在寄存器r0，指令ldmia r0, {r3-r7}把标号1、__pv_table_begin、__pv_table_end、__pv_phys_pfn_offset和__pv_offset的虚拟地址（定义见下面程序最后面）分别读入寄存器r3、r4、r5、r6和r7中。由于MMU还没有运行，此时虚拟地址和物理地址是1：1对应。代码subs r3, r0, r3 得到的是物理地址和虚拟地址的偏差。在运行代码段：

add r4, r4, r3 @ 计算__pv_table的起始位置

add r5, r5, r3 @ 计算__pv_table的末尾位置

add r6, r6, r3 @ 计算__pv_phys_pfn_offset物理地址

add r7, r7, r3 @ 计算__pv_offset物理地址

后，寄存器r4、r5、r6和r7分别保存了标号__pv_table_begin、__pv_table_end、__pv_phys_pfn_offset和__pv_offset的物理地址，即这几个表格的实际存储位置。后续代码把PHYS_OFFSET转换成页码存储到表格__pv_phys_pfn_offset中，而把PHYS_OFFSET-PAGE_OFFSET保存到表格__pv_offset的低端，如PHYS_OFFSET小于PAGE_OFFSET，指令把0xFFFFFFFF保存到表格__pv_offset的高端。指令:

mov r6, r3, lsr #24，teq r3, r6, lsl #24

用于测试物理地址与虚拟地址之间的偏差是否与16MiB对齐，若没有对齐则进行出错处理。

在更新完表格后，程序跳转到fixup_a_pv_table开始打补丁，修改指令的立即数部分。

.text

__fixup_a_pv_table:

adr r0, 3f @计算标号3的物理地址

ldr r6, [r0] @读取__pv_offset的虚拟地址

add r6, r6, r3 @计算__pv_offset的物理地址

ldr r0, [r6, #HIGH_OFFSET] @ pv_offset high word

ldr r6, [r6, #LOW_OFFSET] @ pv_offset low word

mov r6, r6, lsr #24

cmn r0, #1 @判断地址偏差是否为负

moveq r0, #0x400000 @ 若r0值为负，把 r0位22置1

b 2f

1: ldr ip, [r7, r3]

bic ip, ip, #0x000000ff

tst ip, #0xf00 @ 检查指令立即数中rotation 域

orrne ip, ip, r6 @ 如果rotation域非零，修改指令的立即数域的值

biceq ip, ip, #0x400000 @ 如果rotation域为零，清除指令的第22位

orreq ip, ip, r0 @ 依据r0的值，保留原mov指令或把mov指令修改为mvn

str ip, [r7, r3] @把修改后的指令放回原来的位置。

2: cmp r4, r5 @判别是否到__pv_table表格的末尾

ldrcc r7, [r4], #4

bcc 1b

ret lr

ENDPROC(__fixup_a_pv_table)

.align

3: .long __pv_offset

ENTRY(fixup_pv_table)

stmfd sp!, {r4 - r7, lr}

mov r3, #0 @ no offset

mov r4, r0 @ r0 = table start

add r5, r0, r1 @ r1 = table size

bl __fixup_a_pv_table

ldmfd sp!, {r4 - r7, pc}

ENDPROC(fixup_pv_table)

.data

.align 2

.globl __pv_phys_pfn_offset

.type __pv_phys_pfn_offset, %object @变量类型为数据

__pv_phys_pfn_offset:

.word 0

.size __pv_phys_pfn_offset, . -__pv_phys_pfn_offset

.globl __pv_offset

.type __pv_offset, %object

__pv_offset:

.quad 0

.size __pv_offset, . -__pv_offset @大小=当前位置-标号起始位置

下面一段代码的作用是从正确的位置读取地址偏差的高32位和低32位的值，并把低32位偏差右移24位，从而只保留低32位中的高8位值。

adr r0, 3f @计算标号3的物理地址

ldr r6, [r0] @读取__pv_offset的虚拟地址

add r6, r6, r3 @计算__pv_offset的物理地址

ldr r0, [r6, #HIGH_OFFSET] @ pv_offset high word

ldr r6, [r6, #LOW_OFFSET] @ pv_offset low word

mov r6, r6, lsr #24

指令

cmn r0, \#1和moveq r0,#0x400000

的作用是判别地址偏差是否为负数，如为负数，则将r0的值修改为0x400000，用以把mov指令修改为mvn指令。

指令

ldr ip, [r7,r3]

的作用是把指令从_pv_table保存的指令地址读入ip（r12）寄存器。首次执行该指令时，r7保存__pv_table_begin的虚拟地址，r3保存地址偏差值。

指令

bic ip, ip, #0x000000ff

和

tst ip,#0xf00

的作用是清除指令的立即数域中的低8位，判别立即数域中的rotation域是否为0。如果rotation域的值非0，则当前指令为adds，执行指令

orrne ip, ip, r6

修改adds的低8位立即数。如果rotation域为0，则当前指令为mov指令，执行指令

biceq ip, ip, #0x400000和orreq ip, ip, r0

修改mov指令。若此时mov值为0x400000，则指令mov被修改为mvn指令，否则，原mov指令不变，只是把立即数域的低8位清零。最后，通过指令

str ip, [r7, r3]

把修改后指令存储到原来所在的地址。

最后一段代码：

cmp r4, r5

ldrcc r7, [r4], #4

bcc 1b

ret lr

的作用是遍历整个__pv_table，修改所有指令地址保存在该表格内的指令，然后返回调用程序。
